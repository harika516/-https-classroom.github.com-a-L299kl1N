list=[]
def outlier (dataset):
    q1,q3=np.percentile(dataset,[25,75])
    iqr=q3-q1
    lower=q1-(1.5*iqr)
    upper=q3+(1.5*iqr)
    for i in  dataset:
        if((i<lower) or (i>upper)):
            list=i
            return i
            #print("outliers are:",i)
